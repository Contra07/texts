    /// <summary>
    /// Обновляет списки вершин и рёбер в элементах управления (ListBox’ы)
    /// и перерисовывает граф в GraphArea.
    /// </summary>
    private void RefreshLists()
    {
        // Обновляем список вершин
        LstVertices.ItemsSource = null;
        LstVertices.ItemsSource = graph?.Vertices.Select(v => v.Name).ToList();

        // Обновляем список рёбер
        LstEdges.ItemsSource = null;
        LstEdges.ItemsSource = graph?.Edges
            .Select(e => $"{e.Source?.Name}-{e.Target?.Name} : {e.Weight}")
            .ToList();

        // Перерисовываем визуализацию
        UpdateGraphAreaFromModel();
    }

    /// <summary>
    /// Строит визуальный граф в GraphArea на основании текущей модели (graph).
    /// </summary>
    private void UpdateGraphAreaFromModel()
    {
        if (graphArea == null || graphLogic == null || graph == null) return;

        // Генерируем визуальные контролы (VertexControl / EdgeControl)
        try { graphArea.GenerateGraph(true, true); }
        catch { try { graphArea.GenerateGraph(true); } catch { graphArea.RelayoutGraph(); } }

        // Устанавливаем подписи рёбер (веса)
        try
        {
            foreach (var kv in graphArea.EdgesList)
            {
                var edgeControl = kv.Value;
                if (edgeControl == null) continue;
                var dataEdge = edgeControl.Edge as Link;
                if (dataEdge != null)
                {
                    // Ищем элемент TextBlock с именем "PART_EdgeLabel" внутри EdgeControl
                    var label = FindVisualChildByName(edgeControl, "PART_EdgeLabel") as TextBlock;
                    if (label != null) label.Text = dataEdge.Weight.ToString();

                    // Если метки нет — отображаем вес во всплывающей подсказке
                    edgeControl.ToolTip = dataEdge.Weight.ToString();
                }
            }
        }
        catch { /* не критично */ }
    }

    /// <summary>
    /// Рекурсивный поиск визуального элемента по имени в дереве WPF.
    /// Используется для поиска текстовых меток рёбер.
    /// </summary>
    private FrameworkElement FindVisualChildByName(DependencyObject parent, string name)
    {
        if (parent == null) return null;
        for (int i = 0; i < VisualTreeHelper.GetChildrenCount(parent); i++)
        {
            var child = VisualTreeHelper.GetChild(parent, i);
            if (child is FrameworkElement fe && fe.Name == name)
                return fe;
            var result = FindVisualChildByName(child, name);
            if (result != null) return result;
        }
        return null;
    }

        var ptOnZoom = e.GetPosition(ZoomCtrl);

        // Преобразуем координаты в пространство GraphArea (учитывая масштаб и сдвиг)
        Point ptInGraphArea = ptOnZoom;
        if (graphArea != null)
        {
            try
            {
                var relative = graphArea.TransformToVisual(ZoomCtrl);
                ptInGraphArea = relative.Inverse.Transform(ptOnZoom);
            }
            catch
            {
                ptInGraphArea = ptOnZoom;
            }
        }

        // Добавляем вершину в модель
        var newV = graph.AddVertex();
        RefreshLists(); // обновляем визуализацию

        // После генерации ищем VertexControl для новой вершины и задаём позицию
        try
        {
            if (graphArea.VertexList.TryGetValue(newV, out var vc))
            {
                try
                {
                    // Попытка использовать SetPosition (если поддерживается)
                    dynamic dyn = vc;
                    dyn.SetPosition(ptInGraphArea.X - (vc.ActualWidth / 2.0),
                                    ptInGraphArea.Y - (vc.ActualHeight / 2.0));
                }
                catch
                {
                    // Альтернатива: обычный Canvas.SetLeft/Top
                    Canvas.SetLeft(vc, ptInGraphArea.X - (vc.ActualWidth / 2.0));
                    Canvas.SetTop(vc, ptInGraphArea.Y - (vc.ActualHeight / 2.0));
                }
            }
        }
        catch
        {
            // ошибка не критична
        }
    }

CycleEnumerator.cs

using NKM1.Models;
using QuickGraph;

namespace NKM1;

/// <summary>
/// Поиск всех уникальных циклов в графе НКМ и определение их знака.
/// </summary>
public static class CycleDetection
{
    public static List<CycleInfo> FindAllSimpleCycles(
        IBidirectionalGraph<Factor, Link> graph)
    {
        var result = new List<CycleInfo>();
        var visited = new HashSet<Factor>();
        var stack = new Stack<Factor>();
        var currentPath = new HashSet<Factor>();
        var uniqueCycles = new HashSet<string>();

        foreach (var vertex in graph.Vertices)
        {
            DFS(graph, vertex, vertex, visited, stack, currentPath, result, uniqueCycles);
            visited.Add(vertex);
        }

        return result;
    }

    private static void DFS(
        IBidirectionalGraph<Factor, Link> graph,
        Factor startFactor,
        Factor currentFactor,
        HashSet<Factor> visited,
        Stack<Factor> stack,
        HashSet<Factor> currentPath,
        List<CycleInfo> result,
        HashSet<string> uniqueCycles)
    {
        stack.Push(currentFactor);
        currentPath.Add(currentFactor);

        foreach (var edge in graph.OutEdges(currentFactor))
        {
            var next = edge.Target;

            if (next.Equals(startFactor))
            {
                var cycle = stack.Reverse().ToList();
                var key = NormalizeCycle(cycle);

                if (uniqueCycles.Add(key))
                {
                    // вычисляем знак цикла
                    int negativeEdges = CountNegativeEdges(graph, cycle);
                    bool isPositive = negativeEdges % 2 == 0;

                    result.Add(new CycleInfo
                    {
                        Vertices = cycle,
                        IsPositive = isPositive,
                        NegativeEdgeCount = negativeEdges
                    });
                }

                continue;
            }

            if (!visited.Contains(next) && !currentPath.Contains(next))
            {
                DFS(graph, startFactor, next, visited, stack, currentPath, result, uniqueCycles);
            }
        }

        stack.Pop();
        currentPath.Remove(currentFactor);
    }

    /// <summary>
    /// Подсчёт количества отрицательных рёбер в цикле.
    /// </summary>
    private static int CountNegativeEdges(IBidirectionalGraph<Factor, Link> graph, List<Factor> cycle)
    {
        int count = 0;
        for (int i = 0; i < cycle.Count; i++)
        {
            var source = cycle[i];
            var target = cycle[(i + 1) % cycle.Count];

            var edge = graph.Edges.FirstOrDefault(e => e.Source == source && e.Target == target);
            if (edge != null && edge.Weight < 0)
                count++;
        }
        return count;
    }

    /// <summary>
    /// Нормализует цикл — делает его независимым от точки старта и направления.
    /// </summary>
    private static string NormalizeCycle(List<Factor> cycle)
    {
        if (cycle.Count > 1 && cycle.First()?.Equals(cycle.Last()) == true)
            cycle.RemoveAt(cycle.Count - 1);

        var ordered = cycle
            .Select((v, i) => (v, i))
            .OrderBy(x => x.v?.Name)
            .First();

        var normalized = cycle.Skip(ordered.i).Concat(cycle.Take(ordered.i)).ToList();

        string forward = string.Join("→", normalized.Select(v => v.Name));
        string reverse = string.Join("→", normalized.Select(v => v.Name).Reverse());

        return string.Compare(forward, reverse, StringComparison.Ordinal) <= 0
            ? forward
            : reverse;
    }
}

using NKM1.Models;

namespace NKM1;

public class CycleInfo
{
    public List<Factor> Vertices { get; set; } = new();
    public bool IsPositive { get; set; }
    public double NegativeEdgeCount { get; set; }

    public override string ToString()
    {
        var sign = IsPositive ? "Положительный" : "Отрицательный";
        return $"{string.Join(" -> ", Vertices.Select(v => v.Name))} ({sign})";
    }

using System.Numerics;
using MathNet.Numerics.LinearAlgebra;

namespace NKM1.Models;
/// <summary>
/// Класс для анализа графа через матрицу смежности.
/// </summary>
public static class GraphMatrixAnalyzer
{
    /// <summary>
    /// Преобразует граф в матрицу смежности с весами.
    /// </summary>
    /// <param name="graph">Граф (например, NKM)</param>
    /// <returns>Матрица NxN, где N = количество вершин</returns>
    public static double[,] ToAdjacencyMatrix(NKM graph)
    {
        var vertices = graph.Vertices.ToList();
        int n = vertices.Count;
        var matrix = new double[n, n];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
                matrix[i, j] = 0; // по умолчанию нет связи
        }

        foreach (var edge in graph.Edges)
        {
            int i = vertices.IndexOf(edge.Source);
            int j = vertices.IndexOf(edge.Target);
            matrix[i, j] = edge.Weight;
        }

        return matrix;
    }

    /// <summary>
    /// Вычисляет собственные числа матрицы смежности.
    /// </summary>
    public static Complex[] ComputeEigenvalues(double[,] adjacencyMatrix)
    {
        // Используем Math.NET Numerics
        var matrix = Matrix<double>.Build.DenseOfArray(adjacencyMatrix);
        var eigen = matrix.Evd();

        // Возвращаем все собственные значения
        return eigen.EigenValues.Select(c => new Complex(c.Real, c.Imaginary)).ToArray();
    }

    /// <summary>
    /// Находит максимальное по модулю собственное число.
    /// </summary>
    public static Complex GetMaxEigenvalueByMagnitude(Complex[] eigenvalues)
    {
        return eigenvalues.OrderByDescending(e => e.Magnitude).FirstOrDefault();
    }

    /// <summary>
    /// Быстрая связка: от графа до спектрального радиуса.
    /// </summary>
    public static double GetSpectralRadius(NKM graph)
    {
        var matrix = ToAdjacencyMatrix(graph);
        var eigenvalues = ComputeEigenvalues(matrix);
        return eigenvalues.Max(e => e.Magnitude);
    }

    /// <summary>
    /// Пример использования в коде.
    /// </summary>
    public static void ExampleUsage(NKM graph)
    {
        var matrix = ToAdjacencyMatrix(graph);
        var eigenvalues = ComputeEigenvalues(matrix);
        var maxEigen = GetMaxEigenvalueByMagnitude(eigenvalues);

        Console.WriteLine("Матрица смежности:");
        PrintMatrix(matrix);

        Console.WriteLine("\nСобственные числа:");
        foreach (var e in eigenvalues)
            Console.WriteLine($"λ = {e.Real:F3} {(e.Imaginary >= 0 ? "+" : "-")} {Math.Abs(e.Imaginary):F3}i");

        Console.WriteLine($"\nМаксимальное по модулю собственное число: |λmax| = {maxEigen.Magnitude:F3}");
    }

    /// <summary>
    /// Утилита для красивого вывода матрицы в консоль.
    /// </summary>
    public static void PrintMatrix(double[,] matrix)
    {
        int n = matrix.GetLength(0);
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
                Console.Write($"{matrix[i, j],8:F2}");
            Console.WriteLine();
        }
    }
}

using QuickGraph;

namespace NKM1.Models
{
	/// <summary>
    /// Модель графа. Этот граф и является НКМ.
    /// </summary>
	public class NKM : BidirectionalGraph<Factor, Link>
    {
        private int _counter = 0;

        public NKM()
        {

        }

        public Factor AddVertex()
        {
            var v = new Factor("V" + (++_counter));
            base.AddVertex(v);
            return v;
        }

        public void RenameVertex(string oldId, string newId)
        {
            var v = Vertices.FirstOrDefault(x => x.Name == oldId);
            if (v != null) v.Name = newId;
        }


		public void RemoveVertex(string id)
		{
			RemoveVertexIf((v) => v.Name == id);
			RemoveEdgeIf((e) => e.Source?.Name == id || e.Target?.Name == id);
		}

		public bool AddEdge(Factor source, Factor target, double weight)
        {
            if (source is not null && target is not null) 
				return AddEdge(source.ID, target.ID, weight);
			else
				return false;
        }

        public bool AddEdge(long sourceId, long targetId, double weight)
        {
            var s = Vertices.FirstOrDefault(v => v.ID == sourceId);
            var t = Vertices.FirstOrDefault(v => v.ID == targetId);
            if (s != null && t != null && !EdgeExists(sourceId, targetId))
			{
                return base.AddEdge(new Link(s, t, weight));
            }
			else
			{
				return false;
			}
        }


        public void RemoveEdge(string sourceId, string targetId)
        {
            RemoveEdgeIf(e => e.Source?.Name == sourceId && e.Target?.Name == targetId);
        }


		public void SetEdgeWeight(string sourceId, string targetId, double w)
		{
			var e = Edges.FirstOrDefault(ed => ed.Source?.Name == sourceId && ed.Target?.Name == targetId);
			if (e != null) e.Weight = w;
		}
		
		public bool EdgeExists(Factor source, Factor target)
		=>	source is not null && target is not null
			&& EdgeExists(source.ID, target.ID);

		public bool EdgeExists(long sourceId, long targetId)
		=> Edges.Any(e => e.Source.ID == sourceId && e.Target.ID == targetId)
			|| Edges.Any(e => e.Source.ID == targetId && e.Target.ID == sourceId);

		public static NKM CreateRandomDirectedGraph(int vertexCount, int edgeCount, int? randomSeed = null)
		{
			if (vertexCount <= 0)
			{
				throw new ArgumentOutOfRangeException(nameof(vertexCount), "Количество вершин должно быть положительным.");
			}

			// Максимальное количество ребер в простом ориентированном графе
			var maxEdges = vertexCount * (vertexCount - 1);
			if (edgeCount < 0 || edgeCount > maxEdges)
			{
				throw new ArgumentOutOfRangeException(nameof(edgeCount), $"Количество ребер должно быть в диапазоне от 0 до {maxEdges}.");
			}

			var graph = new NKM();
			var random = randomSeed.HasValue ? new Random(randomSeed.Value) : new Random();

			// 1. Создаем вершины и сохраняем их в массив для быстрого доступа
			var vertices = new Factor[vertexCount];
			for (int i = 0; i < vertexCount; i++)
			{
				vertices[i] = graph.AddVertex();
			}

			// 2. Создаем и добавляем случайные ребра
			var existingEdges = new HashSet<Tuple<int, int>>();
			while (graph.EdgeCount < edgeCount)
			{
				// Выбираем случайную исходную и целевую вершины
				var sourceIndex = random.Next(0, vertexCount);
				var targetIndex = random.Next(0, vertexCount);

				// Проверяем, чтобы не было самопетель и дублирующихся ребер
				if (sourceIndex == targetIndex)
				{
					continue;
				}

				var newEdgeTuple = new Tuple<int, int>(sourceIndex, targetIndex);
				if (!existingEdges.Contains(newEdgeTuple))
				{
					var sourceVertex = vertices[sourceIndex];
					var targetVertex = vertices[targetIndex];
					// Создаем ребро со случайным весом
					var weight = (random.Next(0, 2) == 0 ? 1 : -1) * random.NextDouble();
					graph.AddEdge(sourceVertex.ID, targetVertex.ID, weight);

					existingEdges.Add(newEdgeTuple);
				}
			}

			return graph;
		}
	}

        string selected = CmbTarget.SelectedItem.ToString();
        CmbSource.ItemsSource = graph.Vertices
            .Select(v => v.Name)
            .Where(name => name != selected)
            .ToList();
    }

    private void BtnAdd_Click(object sender, RoutedEventArgs e)
    {
        if (CmbSource.SelectedItem == null || CmbTarget.SelectedItem == null)
        {
            MessageBox.Show("Выберите источник и цель.", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
            return;
        }

        if (!double.TryParse(TxtWeight.Text.Replace(',', '.'), System.Globalization.NumberStyles.Float, System.Globalization.CultureInfo.InvariantCulture, out double weight))
        {
            MessageBox.Show("Введите корректное число.", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
            return;
        }

        if (weight < -1 || weight > 1)
        {
            MessageBox.Show("Значение должно быть в диапазоне [-1; 1].", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
            return;
        }


        var sourceName = CmbSource.SelectedItem.ToString();
        var targetName = CmbTarget.SelectedItem.ToString();
        var source = graph.Vertices.First(e => e.Name == sourceName );
        var target = graph.Vertices.First(e => e.Name == targetName);

        if (graph.EdgeExists(source, target))
        {
            MessageBox.Show($"Связь между факторами '{sourceName}' и '{targetName}' ужа существует.", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Warning);
            return;
        }

        Source = source;
        Target = target;
        Weight = weight;

        DialogResult = true;
    }

    private void BtnCancel_Click(object sender, RoutedEventArgs e)
    {
        DialogResult = false;
    }
